Embed Setup Script

Instruct your customers to copy and paste this code injection script right before the closing </body> tag of their checkout page.

<script>
document.addEventListener("DOMContentLoaded", () => {
    const phoneInput = document.querySelector('input[type="tel"]') || document.querySelector('input[name*="phone"]');
    if (phoneInput) {
        phoneInput.addEventListener("blur", async () => {
            if (phoneInput.value.length > 5) {
                await fetch('YOUR_DEPLOYED_APP_URL/api/capture', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        store: window.location.hostname,
                        phone: phoneInput.value,
                        value: document.querySelector('[class*="total"], [id*="total"]')?.innerText || "$49.00"
                    })
                });
            }
        });
    }
});
</script>
